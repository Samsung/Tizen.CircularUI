{
    GroupName = "SoundMode",
    IsSelected = true,
    Value = "sound"
}

var radio2 = new Radio
{
    GroupName = "SoundMode",
    IsSelected = false,
    Value = "vibrate"
}

radio1.Selected += (s,e) =>
{
    Console.WriteLine($"radio1 selected! value:{e.Value}");
};

radio2.Selected += (s,e) =>
{
     Console.WriteLine($"radio2 selected! value:{e.Value}");
};
